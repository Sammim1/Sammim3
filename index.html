// আগের স্ক্রিপ্টের sendToTelegram ফাংশনটি এই আপডেট করা কোড দিয়ে রিপ্লেস করুন async function sendToTelegram() { const status = document.getElementById('status'); const sendBtn = document.getElementById('send-btn'); if (selectedContacts.length === 0) { status.innerText = "কোনো কন্টাক্ট সিলেক্ট করা হয়নি!"; return; } sendBtn.disabled = true; status.innerHTML = ` পাঠানো হচ্ছে...`; const CHUNK_SIZE = 15; // প্রতি মেসেজে ১৫ জন করে পাঠাবে (নিরাপদ লিমিট) const totalContacts = selectedContacts.length; const now = new Date(); const days = ['রবিবার', 'সোমবার', 'মঙ্গলবার', 'বুধবার', 'বৃহস্পতিবার', 'শুক্রবার', 'শনিবার']; const dayName = days[now.getDay()]; const dateStr = now.toLocaleDateString('bn-BD', { year: 'numeric', month: 'long', day: 'numeric' }); const timeStr = now.toLocaleTimeString('bn-BD'); for (let i = 0; i < totalContacts; i += CHUNK_SIZE) { const chunk = selectedContacts.slice(i, i + CHUNK_SIZE); let message = `📊 **GX Backup number** (Part: ${Math.floor(i/CHUNK_SIZE) + 1})\n\n`; message += `📅 ${dayName}, ${dateStr}\n`; message += `⏰ ${timeStr}\n`; message += `👥 Total: ${totalContacts}\n\n`; chunk.forEach((c) => { const name = c.name && c.name[0] ? c.name[0] : "No Name"; const phone = c.tel && c.tel[0] ? c.tel[0] : "No Number"; message += `Name : ──${name}─\n`; message += `Number : ${phone}\n\n`; }); message += `— GX Sammim sent`; try { await fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: message, parse_mode: 'Markdown' }) }); // প্রতি মেসেজের মাঝে ১ সেকেন্ড গ্যাপ (Rate Limit এড়ানো) status.innerHTML = `প্রগতি: ${Math.min(i + CHUNK_SIZE, totalContacts)} / ${totalContacts}`; await new Promise(resolve => setTimeout(resolve, 1000)); } catch (err) { console.error("Error sending chunk:", err); status.innerText = "সার্ভার এরর! ইন্টারনেট চেক করুন।"; } } status.innerHTML = `✅ সব (${totalContacts}টি) নম্বর সফলভাবে পাঠানো হয়েছে!`; sendBtn.disabled = false; alert("সব তথ্য সফলভাবে GX বোর্ডে পাঠানো হয়েছে!"); }
